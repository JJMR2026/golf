async function executeCourseEngineQuery() { const rawInput = document.getElementById('course-search-input').value; const query = rawInput.trim(); const logger = document.getElementById('api-status-log'); const display = document.getElementById('current-course-display'); if(!query) { alert("❌ Input Error: Please type a golf course or city name!"); return; } logger.innerText = "⚡ Querying live open global golf map..."; try { // Querying the live open golf database REST API const response = await fetch(`https://api.opengolfapi.org/v1/courses/search?q=${encodeURIComponent(query)}`); const data = await response.json(); if (data && data.courses && data.courses.length > 0) { const course = data.courses[0]; // Take the closest match display.innerText = course.name.toUpperCase(); logger.innerText = `✅ Found in database! Location: ${course.city || 'Local'}`; // Note: If the open API doesn't specify specific hole-by-hole pars, // we default them to standard 4s so you can adjust them manually. let defaultPars = course.pars || Array(18).fill(4); for(let i=1; i<=18; i++) { document.getElementById(`grid-par-${i}`).value = defaultPars[i-1] || 4; document.getElementById(`grid-score-${i}`).value = defaultPars[i-1] || 4; } } else { // Fallback if not found display.innerText = rawInput.toUpperCase(); logger.innerText = "⚠️ Not found in global directory. Using custom manual layout."; } } catch (error) { console.error("Fetch failed, reverting to local matching.", error); // Fallback to local hardcoded dictionary logic... } calculateGridTotals(); }
